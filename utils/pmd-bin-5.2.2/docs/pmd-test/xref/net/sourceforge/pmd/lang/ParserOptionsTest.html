6   import static org.junit.Assert.assertEquals;
7   import static org.junit.Assert.assertFalse;

12  import org.junit.Test;

14  /**
15   * Unit tests for {@link ParserOptions}.
16   */
17  public class ParserOptionsTest {
18  
19      /**
20       * SuppressMarker should be initially null and changeable.
21       */
22      @Test
23      public void testSuppressMarker() {
24          ParserOptions parserOptions = new ParserOptions();
25          assertNull(parserOptions.getSuppressMarker());
26          parserOptions.setSuppressMarker("foo");
27          assertEquals("foo", parserOptions.getSuppressMarker());
28      }
29  
30      /**
31       * Verify that the equals and hashCode methods work as expected.
32       */
33      @Test
34      public void testEqualsHashcode() {
35          ParserOptions options1 = new ParserOptions();
36          options1.setSuppressMarker("foo");
37          ParserOptions options2 = new ParserOptions();
38          options2.setSuppressMarker("bar");
39          ParserOptions options3 = new ParserOptions();
40          options3.setSuppressMarker("foo");
41          ParserOptions options4 = new ParserOptions();
42          options4.setSuppressMarker("bar");
43          verifyOptionsEqualsHashcode(options1, options2, options3, options4);
44      }
45  
46      /**
47       * Verify equals and hashCode for 4 {@link ParserOptions} instances.
48       * The given options should be as follows: 1 and 3 are equals, as are 2 and 4.
49       * 
50       * @param options1 first option instance - equals third
51       * @param options2 second option instance - equals fourth
52       * @param options3 third option instance - equals first
53       * @param options4 fourth option instance - equals second
54       */
55      public static void verifyOptionsEqualsHashcode(ParserOptions options1, ParserOptions options2,
56              ParserOptions options3, ParserOptions options4) {
57          // Objects should be different
58          assertNotSame(options1, options2);
59          assertNotSame(options1, options2);
60          assertNotSame(options1, options3);
61          assertNotSame(options2, options3);
62          assertNotSame(options2, options4);
63          assertNotSame(options3, options4);
64  
65          // Check all 16 equality combinations
66          assertEquals(options1, options1);
67          assertFalse(options1.equals(options2));
68          assertEquals(options1, options3);
69          assertFalse(options1.equals(options4));
70  
71          assertFalse(options2.equals(options1));
72          assertEquals(options2, options2);
73          assertFalse(options2.equals(options3));
74          assertEquals(options2, options4);
75  
76          assertEquals(options3, options1);
77          assertFalse(options3.equals(options2));
78          assertEquals(options3, options3);
79          assertFalse(options3.equals(options4));
80  
81          assertFalse(options4.equals(options1));
82          assertEquals(options4, options2);
83          assertFalse(options4.equals(options3));
84          assertEquals(options4, options4);
85  
86          // Hashcodes should match up
87          assertNotEquals(options1.hashCode(), options2.hashCode());
88          assertEquals(options1.hashCode(), options3.hashCode());
89          assertNotEquals(options1.hashCode(), options4.hashCode());
90          assertNotEquals(options2.hashCode(), options3.hashCode());
91          assertEquals(options2.hashCode(), options4.hashCode());
92          assertNotEquals(options3.hashCode(), options4.hashCode());
93      }
94  }
